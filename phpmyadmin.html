1. export PHPMYADMIN_VERSION=$(curl --silent https://www.phpmyadmin.net/downloads/ | grep "btn btn-success download_popup" | sed -n 's/.*href="\([^"]*\).*/\1/p' | tr '/' '\n' | grep -E '^.*[0-9]+\.[0-9]+\.[0-9]+$')
2. cd /var/www/pterodactyl/public && wget https://files.phpmyadmin.net/phpMyAdmin/$PHPMYADMIN_VERSION/phpMyAdmin-$PHPMYADMIN_VERSION-all-languages.zip && unzip phpMyAdmin-$PHPMYADMIN_VERSION-all-languages.zip && rm phpMyAdmin-$PHPMYADMIN_VERSION-all-languages.zip && mv phpMyAdmin-$PHPMYADMIN_VERSION-all-languages pma
3. nano /etc/mysql/mariadb.conf.d/50-server.cnf
4. mysql -u root -p -e "GRANT ALL PRIVILEGES ON *.* TO root@'%' IDENTIFIED BY 'anthost' WITH GRANT OPTION;"
5. mysql -u root -p -e "GRANT ALL PRIVILEGES ON *.* TO root@'localhost' IDENTIFIED BY 'anthost' WITH GRANT OPTION;"
6. systemctl restart mysql
